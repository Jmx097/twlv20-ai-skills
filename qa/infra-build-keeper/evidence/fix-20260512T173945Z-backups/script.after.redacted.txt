#!/usr/bin/env bash
set -euo pipefail
umask 077
set -a
source /root/.secrets/b2-backup.env
set +a
STAMP="$(date -u +%Y-%m-%d-%H%MUTC)"
DAY="$(date -u +%Y-%m-%d)"
WORK="$(mktemp -d /var/backups/twlv20-postgres/run.XXXXXX)"
trap 'rm -rf "$WORK"' EXIT
LOG=/var/log/twlv20/postgres-backup.log
DATABASES=(twlv20 infisical)
{
  echo "[$(date -u --iso-8601=seconds)] starting backup batch ${DAY}/${STAMP}"
  for DB in "${DATABASES[@]}"; do
    BASE="${DB}-${STAMP}.dump"
    DUMP="$WORK/$BASE"
    ENC="$DUMP.gpg"
    OBJ="${B2_OBJECT_PREFIX}/${DAY}/${BASE}.gpg"
    echo "[$(date -u --iso-8601=seconds)] dumping ${DB} to custom format"
    sudo -u postgres pg_dump --format=custom --no-owner --no-privileges --dbname="$DB" --file="$DUMP"
    echo "[$(date -u --iso-8601=seconds)] encrypting ${DB} with AES256"
    gpg --batch --yes --symmetric --cipher-algo AES256 --pinentry-mode loopback \
      --passphrase-file <redacted> \
      --output "$ENC" "$DUMP"
    echo "[$(date -u --iso-8601=seconds)] uploading ${OBJ}"
    /usr/local/sbin/twlv20-pg-b2.py upload "$ENC" "$OBJ"
  done
  /usr/local/sbin/twlv20-pg-b2.py retention "$B2_OBJECT_PREFIX/" "$B2_RETENTION_COUNT"
  echo "[$(date -u --iso-8601=seconds)] backup batch complete ${DAY}/${STAMP}"
} | tee -a "$LOG"
